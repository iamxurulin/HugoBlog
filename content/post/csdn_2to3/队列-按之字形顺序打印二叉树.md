### 求解代码

在[【队列】求二叉树的层序遍历](https://blog.csdn.net/qq_44678890/article/details/156832822?spm=1001.2014.3001.5501)的基础上增加一个flag区分奇偶性就行。


```java
public ArrayList<ArrayList<Integer>> Print (TreeNode pRoot) {
        ArrayList<ArrayList<Integer>> ans = new ArrayList<>();

		if(pRoot==null){
			return ans;
		}
		Queue<TreeNode> queue = new ArrayDeque<TreeNode>();
		queue.add(pRoot);

		boolean flag = true;

		while(!queue.isEmpty()){
			ArrayList<Integer> list = new ArrayList<>();

			int n = queue.size();

			flag = !flag;

			for(int i=0;i<n;i++){
				TreeNode cur = queue.poll();
				list.add(cur.val);

				if (cur.left!=null) {
					queue.add(cur.left);
				}
				
				if(cur.right!=null){
					queue.add(cur.right);
				}
			}

			if(flag){
				Collections.reverse(list);
			}
			ans.add(list);
		}

		return ans;
    }
```
